\begin{lstlisting}[style=Matlab]
dt = 0.1    % sample time
t  = 0:dt:14;   % time vektore
u = ones(size(t)); u(1) = 0;    % step input
A = [-2, -2;
      2,  0];
B = [2;
     0];
C = [0.5, 1.25];
D = 0;
x(:,1) = [0;0]; % initial value 0 and time 
dx(:,1) = [0;0];

for i = 2:length(t);    % controller loop
    x(:,i) = x(:,i-1) +dt*dx(:,i-1); 
    dx(:,i) = A*x(:,i-1) + B*u(i);
    y(i) = C*x(:,i) + D*u(i);
end
scatter(t,y,3)  % scatter plot
\end{lstlisting}
